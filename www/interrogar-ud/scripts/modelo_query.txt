import estrutura_ud
import re
conllu = "<!--corpus-->" #N√ÉO EDITAR

def getResultadosBusca():
    corpus = estrutura_ud.Corpus(recursivo=True)
    corpus.load(conllu)

    resultadosBusca = {'output': [], 'casos': 0}

    #EDITAR AQUI
    for sentence in corpus.sentences.values():
        for token in sentence.tokens:
            if token.word == "palavra":
                resultadosBusca['output'].append(sentence.to_str().replace(token.to_str(), "<b>" + token.to_str() + "</b>").replace(sentence.text, re.sub(r"\b" + re.escape(token.word) + r"\b", "<b>" + token.word + "</b>", sentence.text)))

    return resultadosBusca
